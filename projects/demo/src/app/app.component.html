Pipe

Use: 
"{{ '{{' }} list | oxford:'and' }}"
Output:
  • {{ list.length }}: "{{ list | oxford:'and' }}"

Component

Use: 
"<ng-container *ngFor="let x of list; let i = index">
  <a [href]="'https://www.google.com/search?q=' + x">{{ '{{' }} x }}</a>
  <oxford [index]="i" [length]="list.length">or</oxford>
</ng-container>"
Output:
  • {{ list.length }}: " {{ x }} or "
